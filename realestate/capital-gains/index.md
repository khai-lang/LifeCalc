const RULES = {
  basic_deduction: 2500000,
  brackets: [
    {limit:14000000,   rate:0.06, quick:0},
    {limit:50000000,   rate:0.15, quick:1260000},
    {limit:88000000,   rate:0.24, quick:5760000},
    {limit:150000000,  rate:0.35, quick:15440000},

    {limit:1000000000, rate:0.42, quick:35940000},
    {limit:9999999999, rate:0.45, quick:65940000}
  ],
  lthold_default:   [{y:3,r:.06},{y:4,r:.08},{y:5,r:.10},{y:6,r:.12},{y:7,r:.14},{y:8,r:.16},{y:9,r:.18},{y:10,r:.20}],
  lthold_1house:    [{y:3,r:.12},{y:4,r:.16},{y:5,r:.20},{y:6,r:.24},{y:7,r:.28},{y:8,r:.32},{y:9,r:.36},{y:10,r:.40}],
  adj: {unreg:.20, nonbiz:.10, multi:.20}

/* ── 유틸 ── */
function fmtCgt(el){
  const r=el.value.replace(/[^0-9]/g,'');
  el.value=r?parseInt(r).toLocaleString('ko-KR'):'';
  el.dataset.raw=r;
}
function gV(id){
  const el=document.getElementById(id);
  if(!el) return 0;
  // 숫자 인풋 (type=number)
  if(el.type==='number') return parseFloat(el.value)||0;
  // 텍스트 (raw 저장)
  const raw=el.dataset.raw||el.value.replace(/[^0-9.]/g,'');
  return parseFloat(raw)||0;
}
function fw(n){return Math.round(n).toLocaleString('ko-KR')+'원';}
function ck(id){return document.getElementById(id)?.checked||false;}

function yearsBetween(s,e){
  if(!s||!e) return 0;
  const sd=new Date(s),ed=new Date(e);
  if(isNaN(sd)||isNaN(ed)||ed<=sd) return 0;
  return (ed-sd)/(1000*60*60*24*365.25);
}

function autoHolding(){
  const y=yearsBetween(
    document.getElementById('buyDate').value,
    document.getElementById('sellDate').value
  );
  const msg=document.getElementById('holdingAutoMsg');
  if(y>0){
    document.getElementById('holdingYearsManual').value=y.toFixed(2);
    msg.textContent=`보유기간 자동 계산: ${y.toFixed(2)}년`;
  } else {
    msg.textContent='';
  }
}

/* ── 장기보유특별공제율 ── */
function getLtholdRate(assetType, years, isOneHome){
  if(years<3) return 0;
  const isSpecial=(assetType==='house'||assetType==='luxury')&&isOneHome;
  const table=isSpecial?RULES.lthold_1house:RULES.lthold_default;
  let rate=0;
  table.forEach(row=>{if(years>=row.y) rate=row.r;});
  return rate;
}

/* ── 누진세 ── */
function progressiveTax(base){
  if(base<=0) return 0;
  const row=RULES.brackets.find(b=>base<=b.limit)||RULES.brackets[RULES.brackets.length-1];
  return Math.max(0,Math.round(base*row.rate-row.quick));
}

/* ── 메인 계산 ── */
window.calcCGT=function(){
  const assetType=document.getElementById('assetType').value;
  const buy=gV('buyPrice'), sell=gV('sellPrice'), expense=gV('expense');
  const isUnreg=ck('isUnregistered'), isOneHome=ck('isOneHome');
  const isMulti=ck('isMultiHomeHeavy'), isNonBiz=ck('isNonBusinessLand');
  const isTaxExempt=ck('isTaxExemptReview'), isAdjArea=ck('isAdjustedArea');
  const isInherited=ck('isInherited'), isLongRental=ck('isLongTermRental');

  let basicDed=isUnreg?0:(gV('basicDeduction')||RULES.basic_deduction);

  const holdingYears=gV('holdingYearsManual')||0;
  const gain=Math.max(0,sell-buy-expense);

  // 단기 보유 고정 세율
  if(holdingYears>0&&holdingYears<1){
    return showResult(assetType,holdingYears,gain,0,0,gain,gain*0.70,0,'단기(1년 미만) 70% 단일세율',[
      '1년 미만 보유: 70% 단일세율이 적용됩니다. 기본공제·장특공제 없습니다.'
    ],sell,isOneHome);
  }
  if(holdingYears>=1&&holdingYears<2){
    return showResult(assetType,holdingYears,gain,0,0,gain,gain*0.60,0,'단기(2년 미만) 60% 단일세율',[
      '1~2년 미만 보유: 60% 단일세율이 적용됩니다.'
    ],sell,isOneHome);
  }

  // 장특공제
  const ltRate=isUnreg?0:getLtholdRate(assetType,holdingYears,isOneHome);
  const lthold=Math.round(gain*ltRate);
  const incomeAmt=gain-lthold;
  const taxBase=Math.max(0,incomeAmt-basicDed);

  // 가산세율
  let addRate=0, notices=[];
  if(isUnreg){addRate+=RULES.adj.unreg;notices.push('미등기 양도: 기본세율+20%p 가산, 기본공제 없음.');}
  if(isNonBiz&&assetType==='land'){addRate+=RULES.adj.nonbiz;notices.push('비사업용 토지: 기본세율+10%p 가산.');}
  if(isMulti&&(assetType==='house'||assetType==='luxury')){addRate+=RULES.adj.multi;notices.push('다주택 중과(조정지역): 기본세율+20%p 가산. 실제 적용 여부를 공식 확인하세요.');}
  if(isAdjArea) notices.push('조정대상지역: 실제 지역 지정 여부를 반드시 확인하세요.');
  if(isTaxExempt||isOneHome) notices.push('1세대 1주택 비과세·특례는 요건 충족 여부를 별도 확인하세요.');
  if(isInherited) notices.push('상속 자산은 피상속인 취득일·가액 적용 등 별도 규정이 있습니다.');
  if(isLongRental) notices.push('장기임대주택 특례는 임대 기간·면적·가액 등 세부 요건을 확인하세요.');
  if(!buy||!sell) notices.push('취득가액과 양도가액을 입력하면 더 정확한 결과를 확인할 수 있습니다.');

  const baseTax=progressiveTax(taxBase);
  const extraTax=Math.round(taxBase*addRate);
  const grossTax=baseTax+extraTax;

  showResult(assetType,holdingYears,gain,lthold,ltRate,taxBase,grossTax,basicDed,`누진세율 + 가산 ${addRate>0?(addRate*100).toFixed(0)+'%p':'없음'}`,notices,sell,isOneHome);
};
